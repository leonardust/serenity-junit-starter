        // Get references to the login and password input fields

        const loginInput = document.getElementById("login_id");

        const passwordInput = document.getElementById("login_password");



        // Get a reference to the login button

        const loginButton = document.getElementById("login-btn");



        // Add event listeners to the input fields to check if they have at least 8 characters

        loginInput.addEventListener("input", checkInputs);

        passwordInput.addEventListener("input", checkInputs);

        loginButton.addEventListener("click", login);



        function checkInputs() {

          // Check if both input fields have at least 8 characters

          if (loginInput.value.length >= 8 && passwordInput.value.length >= 8) {

            // If so, enable the login button

            loginButton.disabled = false;

          } else {

            // If not, disable the login button

            loginButton.disabled = true;

          }

        }



        function login() {

            setCookie("isLogged", 'true', 1);

            window.location.href = 'pulpit.html';

        }



        function setCookie(cname, cvalue, exdays) {

            const d = new Date();

            d.setTime(d.getTime() + (exdays*24*60*60*1000));

            let expires = "expires="+ d.toUTCString();

            document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";

        }
