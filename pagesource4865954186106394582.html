        function getParameterByName(name) {

            name = name.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");

            var regex = new RegExp("[\\?&]" + name + "=([^&#]*)"),

                results = regex.exec(location.search);

            return results == null ? "" : decodeURIComponent(results[1].replace(/\+/g, " "));

        }



        function getParam(name) {

            name = name.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");

            var regexS = "[\\?&]" + name + "=([^&#]*)";

            var regex = new RegExp(regexS);

            var results = regex.exec(decodeURIComponent(window.location.href));

            if (results == null)

                return "Brak wiadomości";

            else

                return results[1];

        }

            var transfer_amount = document.getElementById("widget_1_transfer_amount").value

            var transfer_title = document.getElementById("widget_1_transfer_title").value



            if (transfer_title.length > 0 && transfer_amount.length > 0 && transfer_receiver.length > 0) {

                document.getElementById("widget_1_transfer_title").value = ''



                var amount_before_raw = document.getElementById("money_value").innerText

                var decimal_before_raw = document.getElementById("decimal_value").innerText



                amount_before_raw = amount_before_raw.replaceAll(' ', '').replaceAll(' ', '').replaceAll(',', '')

                const value = (parseInt(amount_before_raw) * 100) + parseInt(decimal_before_raw)

                transfer_amount_decimal = transfer_amount_decimal !== undefined ? transfer_amount_decimal : 0



                var result = value - (parseInt(transfer_amount_money) * 100 + parseInt(transfer_amount_decimal))

                var result_str = `${result}`

                window.manager.require('app:modules:forms:submitForm:clearErrors')

            }

        }
